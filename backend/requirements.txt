# Version ranges (not pins) so the same file installs cleanly on Python
# 3.10 – 3.14. Several of the previously-pinned versions ship no wheel
# for Python 3.13/3.14, and on Windows without a Rust toolchain that
# means install failure → backend never starts → frontend shows
# "Federal data unavailable". Floors are the earliest versions known to
# work; ceilings are the next major where we'd expect breaking changes.
fastapi>=0.115,<0.120
uvicorn[standard]>=0.48.0,<1
httpx>=0.27,<0.30
python-dotenv>=1.0,<2.0
pydantic>=2.11,<3
pydantic[email]>=2.11,<3

# now calls bcrypt directly, which works on bcrypt 4.x and 5.x.
sqlalchemy>=2.0.35,<3
bcrypt>=4.3,<6
itsdangerous>=2.2,<3
# Required by FastAPI when endpoints accept UploadFile / multipart form
# data — specifically the post-image upload endpoint.

# Postgres driver — only used in production deployments where
# DATABASE_URL points at a Postgres server. Local dev keeps using
# SQLite (no extra dep needed; SQLite ships with Python). The
# `-binary` distribution avoids needing libpq + a C toolchain on
# the build machine, which Render's free build tier doesn't have.
psycopg2-binary>=2.9,<3
# Anthropic API client — powers AI features (comment classification,
# post summaries, etc.). The client is initialized lazily inside
# app/services/ai_service.py so a missing ANTHROPIC_API_KEY at boot
# just degrades the AI surfaces to "Coming soon" without taking the
# whole API down. Version floor 0.40 is the first release with
# native streaming + citations support; ceiling is the next major.
anthropic>=0.40,<1

# Task #62 — TOTP 2FA for admin / rep / candidate accounts (citizens
# optional). pyotp generates + verifies the 6-digit time-based codes;
# cryptography (Fernet) symmetrically encrypts the per-account TOTP
# secret at rest, keyed off SESSION_SECRET via HKDF so the encrypted
# secret on disk is useless without the env var. Recovery codes are
# bcrypt-hashed (no new dep — bcrypt already above for passwords).
pyotp>=2.9,<3
cryptography>=43,<46

# Task #83 — Cloudflare R2 object storage for post images (replaces
# the ephemeral local disk path on Render). boto3 talks to R2 via
# R2's S3-compatible API; the storage abstraction in
# services/image_storage.py imports it lazily, so dev environments
# without R2 credentials don't crash on missing creds — they fall
# back to LocalDiskStorage. Floor 1.34 is the first release with
# native support for the SigV4 auth R2 expects.
boto3>=1.34,<2

# Task #87 — Postmark transactional email (password resets, account
# deletion confirmations, future subscription receipts). postmarker
# is the most-maintained Python client. Lazy-imported inside
# PostmarkEmailService.__init__ so dev environments without
# POSTMARK_API_TOKEN don't crash on import — they fall back to
# DevEmailService which logs to stdout. Floor 1.0 is the stable
# release; ceiling at 2 catches breaking-API rewrites.
postmarker>=1.0,<2

# Task #88 — Stripe subscriptions ($5/mo citizen tier). Lazy-imported
# inside StripeBillingService.__init__ so dev environments without
# STRIPE_API_KEY don't crash on import — they fall back to
# DevBillingService which simulates the checkout/webhook flow without
# touching Stripe. Floor 7.0 is the first release with the stable
# Customer Portal API surface; ceiling at 13 leaves room for the
# in-progress 12.x line.
stripe>=7.0,<13
defusedxml>=0.7.1

# Push notifications (FCM) — firebase-admin sends device pushes +

firebase-admin>=6.5,<8
